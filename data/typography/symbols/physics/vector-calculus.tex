% ****************************************
% Vector Calculus

% * general vector style *
\def\vect#1{\bold{#1}}
\def\unitvect#1{\vect{\hat#1}}

% * unit coordinate vectors *
\def\xhat{\unitvect{x}}
\def\yhat{\unitvect{y}}
\def\zhat{\unitvect{z}}
\def\rhat{\unitvect{r}}
\def\shat{\unitvect{s}}
\def\nhat{\unitvect{n}}
\def\thetahat{\unitvect{\theta}}
\def\phihat{\unitvect{\phi}}

% * vector derivative operators *
\def\diver#1{\nabla \cdot #1}
\def\curl#1{\nabla \times #1}
\def\dirderiv#1{#1 \cdot \nabla}

% * partial derivative operator *
\def\dee#1{\partial_{#1}}
